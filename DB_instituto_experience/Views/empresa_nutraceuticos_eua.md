with `transacoes_cartpanda` as (select `cp`.`created_at_date` AS `created_at_date`,sum(coalesce(`cp`.`total_price`,0)) AS `total_price`,sum(coalesce(`cp`.`commission`,0)) AS `commission`,sum(coalesce(`cp`.`taxes`,0)) AS `taxes`,sum(coalesce(`cp`.`total_refund`,0)) AS `total_refund`,sum((case when (`cp`.`payment_status` in ('approved','refunded_partial')) then `custo_produto_cartpanda`,coalesce(`cc`.`total_price`,0) AS `receita_call_center`,coalesce(`cc`.`commission`,0) AS `comissao_call_center`,coalesce(`cc`.`taxes`,0) AS `receita_total`,(coalesce(`cp`.`commission`,0) + coalesce(`cc`.`commission`,0)) AS `comissao_total`,(coalesce(`cp`.`taxes`,0) + coalesce(`cc`.`taxes`,0)) AS `comissao_cartpanda`,`dados_consolidados`.`taxes_cartpanda` AS `taxes_cartpanda`,`dados_consolidados`.`reembolso_cartpanda` AS `reembolso_cartpanda`,`dados_consolidados`.`custo_produto_cartpanda` AS ((((`dados_consolidados`.`comissao_total` - `dados_consolidados`.`custo_produto_total`) - `dados_consolidados`.`gasto_ads_total`) - ((`dados_consolidados`.`receita_cartpanda` * 0.15) - `dados_consolidados`.`reembolso_total`)) > 0) then ((((`dados_consolidados`.`comissao_total` - `dados_consolidados`.`custo_produto_total`) - `dados_consolidados`.`gasto_ads_total`) -
